包管理工具

包管理工具 – 需要连接到仓库，安装仓库中的版本，无法获取最新；阿里云服务器的仓库是阿里云内部的创建的yum仓库
yum search - 搜索仓库中的软件包
yum -y install - 安装软件包
yum remove - 删除软件包
yum upgrade -更新软件包
yum update - 更新所有的包
yum list installed - 列出已经安装过的包
yum info - 查看包的信息

rpm — Redhat Package Manager
rmp -ivh .rpm
rpm -e
rpm -qa | grep mysql

例如：yum install nginx

#源代码构建安装
1.wget 下载源代码
2.解压缩和解归档
gunzip / xz -d
tar -xvf / tar -cvf
3.添加必要的依赖项 - yum
4.构建前的配置 - configure –> Makefile
5.构建和安装 – make && make install
make clean(如果第一次make失败了)
make
make install
6.配置必须要的环境变量 - export / 修改配置
7.注册快捷方式

注意：步骤3 步骤6和步骤7可以依据具体软件删减

#装版本控制软件 – git

  1. wget https://mirrors.edge.kernel.org/pub/software/scm/git/git-2.21.0.tar.xz
  2. xz -d git-2.21.0.tar.xz – >解压
  3. tar -xvf git-2.21.0.tar
  4. cd git-2.21.0
  5. yum -y install libcurl-devel
  6. ./configure –prefix=/usr/local
  7. make && make install
  8. git –version

#装数据库

  1. wget http://download.redis.io/releases/redis-5.0.5.tar.gz
  2. gunzip -d redis-5.0.5.tar.gz
  3. tar -xvf redis-5.0.5.tar
  4. cd redis-5.0.5
  5. make && make install
  6. redis-server –version
  7. redis-cli –version

#安装python

  1. wget https://www.python.org/ftp/python/3.7.3/Python-3.7.3.tgz
  2. gunzip -d python-3.7.3.tgz
  3. tar -xvf python-3.7.3.tar
  4. yum -y install zlib-devel bzip2-devel openssl-devel ncurses-devel sqlite-devel readline-devel tk-devel gdbm-devel libdb4-devel libpcap-devel xz-devel libffi-devel
  5. cd Python3.7.3
  6. ./configure –prefix=/usr/local/python37 –enable-optimizations
  7. make && make install
  8. echo $PATH
  9. export PATH=$PATH:/usr/local/python37/bin
  10. ln -s /usr/local/python37/bin/python3 /usr/bin/python3

##操作解释

  1. 安装依赖项
    yum -y install zlib-devel bzip2-devel openssl-devel ncurses-devel sqlite-devel readline-devel tk-devel gdbm-devel libdb4-devel libpcap-devel xz-devel libffi-devel
    -y 所有询问都回答yes

  2. 安装前的配置 –> 就是设置安装路径
    ./configure –prefix=/usr/local/python37 –enable-optimizations
    –prefix –>指定安装路径

  3. 注册环境变量 （临时）–
    echo $PATH – > 查看当前的环境变量
    export PATH=$PATH:/usr/local/python37/bin
    –> 导出安装路径的原来的值并把路径赋给PATH
    export –>导出

  4. 注册快捷方式：ln – link的简写
    给python3做一个快捷方式（符号链接/软链接）

ln -s /usr/local/python37/bin/python3 /usr/bin/python3
-s(symbolic)
快捷方式放在/usr/bin/python3

ln sohu.html /sohu3.html
创建硬链接 – 备份
删除所有引用 – 就是删除所有数据

  1. gnu zip –> gzip –>压缩文件后缀gz
    gcc – >g协议下的c compiler
    gzip压缩 – gunzip解压
    tar -xvf 文件名字 –>
    -xvf –抽取文件归档 （x –> extract）
    -cvf – 创建文件归档

#mysql

  1. wget https://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-5.7.26-1.el7.x86_64.rpm-bundle.tar

  2. tar -xvf mysql-5.7.26-1.el7.x86_64.rpm-bundle.tar

  3. yum list installed | grep mariadb

  4. yum erase mariadb-libs

  5. rpm -ivh mysql-community-common-5.7.26-1.el7.x86_64.rpm
    rpm -ivh mysql-community-libs-5.7.26-1.el7.x86_64.rpm
    rpm -ivh mysql-community-client-5.7.26-1.el7.x86_64.rpm
    yum install libaio
    rpm -ivh mysql-community-server-5.7.26-1.el7.x86_64.rpm

  6. systemctl start mysqld – > 启动服务

  7. cd /var/log –>查看密码

  8. cat mysqld.log | grep password –> rYf)TJJ;0?ti

  9. mysql -u root -p –>登陆

  10. set global validate_password_policy=0; –>修改密码强度问题
    set global validate_password_length=6; –>修改密码长度问题
    alter user ‘root’@’localhost’ identified by ‘123456’; –> 修改密码为123456

  11. ->创建远程连接账号和密码:
    create user ‘root’@’%’ identified by ‘123456’;
    授权
    grant all privileges on . to ‘root’@’%’ with grant option;
    让权限生效
    flush privileges;

注意：
systemctl start mysqld –> daemon
